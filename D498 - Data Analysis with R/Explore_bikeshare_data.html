#----------------------------------------
# Helper Function: Load Data
#----------------------------------------
load_data <- function(city) {
  # Build the file name based on the city name (e.g., "chicago.csv")
  file_name <- paste0(city, ".csv")
  # Read the CSV file; note stringsAsFactors = FALSE avoids unwanted factor conversion
  data <- read.csv(file_name, stringsAsFactors = FALSE)
  
  # Convert the Start Time and End Time columns into date/time objects
  # Adjust the format if needed (here we assume "YYYY-MM-DD HH:MM:SS")
  data$Start.Time <- as.POSIXct(data$Start.Time, format="%Y-%m-%d %H:%M:%S")
  data$End.Time   <- as.POSIXct(data$End.Time, format="%Y-%m-%d %H:%M:%S")
  
  return(data)
}

# Analysis Function: Popular Times of Travel
#----------------------------------------
popular_times <- function(data) {
  # Create new columns for month, day of week, and hour from the start time
  data <- data %>%
    mutate(month = month(Start.Time, label = TRUE, abbr = FALSE),
           day   = wday(Start.Time, label = TRUE, abbr = FALSE),
           hour  = hour(Start.Time))
  
  # Create frequency tables
  month_count <- table(data$month)
  day_count   <- table(data$day)
  hour_count  <- table(data$hour)
  
  # Find the most common month, day, and hour
  popular_month <- names(which.max(month_count))
  popular_day   <- names(which.max(day_count))
  popular_hour  <- names(which.max(hour_count))
  
  cat("Most common month: ", popular_month, "\n")
  cat("Most common day:   ", popular_day, "\n")
  cat("Most common start hour: ", popular_hour, "\n")
  
  # ----------------------------
  # Plotting the popular times:
  # ----------------------------
  
  # Base R plots 
  par(mfrow=c(1,3))  # Set up a 1x3 grid of plots
  
  barplot(month_count,
          main = "Trips by Month",
          xlab = "Month",
          ylab = "Number of Trips",
          col  = "skyblue",
          las  = 2)
  
  barplot(day_count,
          main = "Trips by Day of Week",
          xlab = "Day",
          ylab = "Number of Trips",
          col  = "salmon",
          las  = 2)
  
  barplot(hour_count,
          main = "Trips by Start Hour",
          xlab = "Hour of Day",
          ylab = "Number of Trips",
          col  = "lightgreen")
  
  par(mfrow=c(1,1))  # Reset plot layout
}

# Analysis Function: Popular Stations and Trips
#----------------------------------------
popular_stations <- function(data) {
  # Compute frequency tables for start and end stations
  start_station_count <- table(data$Start.Station)
  end_station_count   <- table(data$End.Station)
  
  # Identify the most common start and end stations
  popular_start_station <- names(which.max(start_station_count))
  popular_end_station   <- names(which.max(end_station_count))
  
  # Create a combined trip column (e.g., "Station A to Station B")
  data$trip <- paste(data$Start.Station, "to", data$End.Station)
  trip_count <- table(data$trip)
  popular_trip <- names(which.max(trip_count))
  
  cat("Most common start station: ", popular_start_station, "\n")
  cat("Most common end station:   ", popular_end_station, "\n")
  cat("Most common trip:          ", popular_trip, "\n")
  
  # Plot the top 10 most common trips
  top10_trips <- sort(trip_count, decreasing = TRUE)[1:10]
  barplot(top10_trips,
          main = "Top 10 Most Common Trips",
          xlab = "Trip (Start to End)",
          ylab = "Count",
          col  = "purple",
          las  = 2)  # Rotate x-axis labels for readability
}

trip_duration_stats <- function(data) {
  # Compute total and average trip duration (in seconds)
  total_travel_time <- sum(data$Trip.Duration, na.rm = TRUE)
  mean_travel_time  <- mean(data$Trip.Duration, na.rm = TRUE)
  
  cat("Total travel time (in seconds): ", total_travel_time, "\n")
  cat("Average travel time (in seconds): ", mean_travel_time, "\n")
  
  # Plot: Distribution of Trip Durations
  hist(data$Trip.Duration,
       breaks = 50,
       main   = "Trip Duration Distribution",
       xlab   = "Trip Duration (seconds)",
       col    = "orange",
       border = "white")
}
